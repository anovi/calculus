# Compound savings growth with recurring contributions

## Set your params
principal: 1000 EUR //initial
contribution: 100 EUR //each
contributions_per_year: 12
years: 5
annual_rate: 0.06
"How often interest is compounded (12 - monthly, 4 - quarterly, 1 - yearly)"
compounding_frequency: 12

## Calculation (skip)
comp_period_rate: annual_rate / compounding_frequency
comp_periods: compounding_frequency * years
principal_growth: (1 + comp_period_rate) ^ comp_periods

contrib_intervals: compounding_frequency / contributions_per_year
contrib_growth_per_interval: (1 + comp_period_rate) ^ contrib_intervals
contrib_future: contribution * (principal_growth - 1) / (contrib_growth_per_interval - 1)

## See the result
future_value: principal * principal_growth + contrib_future
